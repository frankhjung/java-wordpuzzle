1   package com.marlo.games;

3   import com.pholser.junit.quickcheck.generator.GenerationStatus;
4   import com.pholser.junit.quickcheck.generator.Generator;
5   import com.pholser.junit.quickcheck.random.SourceOfRandomness;
6   
7   /** Generate alpha-numeric characters. */
8   public final class AlphabeticGenerator extends Generator<Character> {
9   
10    /**
11     * Lowercase alphabetic characters: "a" to "z".
12     *
13     * <p>Compare with: <code>
14     *   IntStream.rangeClosed('a', 'z')
15     *      .mapToObj(c -> "" + (char) c)
16     *      .collect(Collectors.joining());
17     * </code>
18     */
19    private static final String ALPHABETIC = "abcdefghijklmnopqrstuvwxyz";;
20  
21    /** Inherit form super class. */
22    public AlphabeticGenerator() {
23      super(Character.class);
24    }
25  
26    /** Generate an alphabetic character. */
27    @Override
28    public Character generate(final SourceOfRandomness randomness, final GenerationStatus status) {
29      return ALPHABETIC.charAt(randomness.nextInt(ALPHABETIC.length()));
30    }
31  }
