1   package com.marlo.games;
2   
3   import java.util.Set;
4   import javax.validation.ConstraintViolation;
5   import javax.validation.Validation;
6   import javax.validation.Validator;
7   import javax.validation.constraints.Max;
8   import javax.validation.constraints.Min;
9   import javax.validation.constraints.Pattern;
10  import lombok.Getter;
11  import lombok.Setter;
12  
13  /** Bean to validate Word Puzzle command line arguments. */
14  @Getter
15  @Setter
16  public class WordPuzzleBean {
17  
18    /** Size validation message. */
19    public static final String SIZE_MESSAGE = "Size: Expected integer in range 1..9";
20    /** Mandatory validation message. */
21    public static final String MANDATORY_MESSAGE = "Mandatory: Expected 1 alphabetic letter";
22    /** Letters validation message. */
23    public static final String LETTERS_MESSAGE = "Letters: Expected 9 alphabetic letters";
24  
25    /** Minimum word size. */
26    @Min(value = 1, message = SIZE_MESSAGE)
27    @Max(value = 9, message = SIZE_MESSAGE)
28    private int size;
29  
30    /** Mandatory character. Every word must contain this character. */
31    @Pattern(regexp = "^[a-z]$", message = MANDATORY_MESSAGE)
32    private String mandatory;
33  
34    /** Words must contain only these letters. */
35    @Pattern(regexp = "^[a-z]{9}$", message = LETTERS_MESSAGE)
36    private String letters;
37  
38    /** Bean validations if any. */
39    private Set<ConstraintViolation<WordPuzzleBean>> validations;
40  
41    /** Construct bean from command line arguments. */
42    public WordPuzzleBean(final int size, final String mandatory, final String letters) {
43      this.size = size;
44      this.mandatory = mandatory;
45      this.letters = letters;
46    }
47  
48    /** Validate the values of this bean. */
49    public boolean validate() {
50      final Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
51      validations = validator.validate(this);
52      return validations.isEmpty();
53    }
54  }
